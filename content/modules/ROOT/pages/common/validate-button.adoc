<script>
(function() {
  function initializeValidateButtons() {
    document.querySelectorAll('.validate-button-placeholder').forEach(placeholder => {
      if (placeholder.classList.contains('initialized')) return;

      const module = placeholder.dataset.module;
      if (!module) return;

      const container = document.createElement('div');
      container.className = 'validate-button-container';

      const button = document.createElement('button');
      button.className = 'validate-button';
      button.innerHTML = `
        <span class="spinner"></span>
        <span class="text">✓ Validate Module</span>
      `;

      const output = document.createElement('pre');
      output.className = 'validate-output';

      container.appendChild(button);
      container.appendChild(output);
      placeholder.appendChild(container);
      placeholder.classList.add('initialized');

      button.addEventListener('click', async () => {
        button.disabled = true;
        button.classList.add('loading');
        button.querySelector('.text').textContent = 'Validating...';
        output.textContent = '';
        output.classList.remove('show');

        try {
          const response = await fetch(\`/stream/validate/\${module}\`);
          const reader = response.body.getReader();
          const decoder = new TextDecoder();

          output.classList.add('show');

          while (true) {
            const {done, value} = await reader.read();
            if (done) break;

            const chunk = decoder.decode(value, {stream: true});
            const lines = chunk.split('\\n');

            for (const line of lines) {
              if (line.startsWith('data: ')) {
                const data = line.slice(6);
                if (data === '[DONE]') continue;
                output.textContent += data + '\\n';
                output.scrollTop = output.scrollHeight;
              }
            }
          }

          button.querySelector('.text').textContent = '✓ Validation Complete';
          setTimeout(() => {
            button.querySelector('.text').textContent = '✓ Validate Module';
          }, 3000);

        } catch (error) {
          output.textContent = 'Error: ' + error.message;
          output.classList.add('show', 'error');
          button.querySelector('.text').textContent = '❌ Failed';
        } finally {
          button.disabled = false;
          button.classList.remove('loading');
        }
      });
    });
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', initializeValidateButtons);
  } else {
    initializeValidateButtons();
  }

  // Re-initialize on page navigation
  const observer = new MutationObserver(initializeValidateButtons);
  observer.observe(document.body, { childList: true, subtree: true });
})();
</script>
